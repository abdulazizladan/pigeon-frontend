Base URL
/ticket

1. Ticket Management
Create Ticket
Method: POST
Path: /
Roles: Director, Manager
Body: 
CreateTicketDto
typescript
interface CreateTicketDto {
  title: string;       // Required
  description?: string; // Optional
  status?: 'active' | 'resolved' | 'terminated' | 'canceled'; // Default: 'active'
}
Get All Tickets
Method: GET
Path: /
Roles: Admin
Response: Ticket[]
Get Ticket by ID
Method: GET
Path: 
/:id
Roles: Admin, Director, Manager
Response: 
Ticket
Get Tickets by Email
Method: GET
Path: /email/:email
Roles: Director, Manager
Response: Ticket[]
Update Ticket
Method: PATCH
Path: 
/:id
Roles: Admin
Body: 
UpdateTicketDto
 (Partial 
CreateTicketDto

typescript
interface UpdateTicketDto {

Delete Ticket
Method: DELETE
Path: 
/:id

typescript

2. Ticket Replies

Method: POST
Path: /:ticketID/reply
Roles: Admin, Director, Manager
Body: 
CreateReplyDto
typescript
interface CreateReplyDto {
  message: string;      // Required
  status?: 'active' | 'resolved' | 'terminated' | 'canceled'; // Optional
}
Response: 
Reply
 (Contains the new reply object)
